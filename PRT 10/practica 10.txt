public class CicloWhile{
  public static void main(String args[]){

 int i = 1, o = 10, u = -5, e = 2, a = 625;

   while(i < 10){
    System.out.print(i + ", ");
    i+=2;

   } System.out.println("");

   while(o < 60){
    System.out.print(o + ", ");
    o+=10;

   } System.out.println("");

   while(u < 0){
    System.out.print(u + ", ");
    u++;

   } System.out.println("");

   while(e < 33){
    System.out.print(e + ", ");
    e = e * 2;

   } System.out.println("");

   while(a >= 1){
    System.out.print(a + ", ");
    a = a / 5;

   } System.out.println("");

    System.out.println("Fin");

 }
}
